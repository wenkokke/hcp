    \AXC{$\seq[{ P }]{
        \mathcal{G}\hsep
        \Gamma, \tmty{x}{A} \hsep
        \Delta, \tmty{x}{A^\bot}
      }$}
    \NOM{Cut}

        \mathcal{G} \hsep \mathcal{H} }$}
  \end{prooftree*}}
