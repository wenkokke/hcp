\begin{lemma}\label{lem:hccp-preservation-subst}
  If $\seq[P]{\mathcal{G} \hsep \tmty{x}{A}, \Gamma}$,
  then $\seq[\cpSub{y}{x}{P}]{\mathcal{G} \hsep \tmty{y}{A}, \Gamma}$.
\end{lemma}
\begin{proof}
  By induction on the structure of the premise.
\end{proof}
